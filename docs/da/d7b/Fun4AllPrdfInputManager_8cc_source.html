237 Fun4AllPrdfInputManager::PushBackEvents(const int i)
238 {
239  // PushBackEvents is supposedly pushing events back on the stack which works
240  // easily with root trees (just grab a different entry) but hard in these HepMC ASCII files.
241  // A special case is when the synchronization fails and we need to only push back a single
242  // event. In this case we save the evt pointer as save_evt which is used in the run method
243  // instead of getting the next event.
244  if (i > 0)
245  {
246  if (i == 1 && evt) // check on evt pointer makes sure it is not done from the cmd line
247  {
248  save_evt = evt;
249  return 0;
250  }
251  cout << PHWHERE << ThisName
252  << " Fun4AllPrdfInputManager cannot push back " << i << " events into file"
253  << endl;
254  return -1;
255  }
256  if (!eventiterator)
257  {
258  cout << PHWHERE << ThisName
259  << " no file open" << endl;
260  return -1;
261  }
262  // Skipping events is implemented as
263  // pushing a negative number of events on the stack, so in order to implement
264  // the skipping of events we read -i events.
265  int nevents = -i; // negative number of events to push back -> skip num events
266  int errorflag = 0;
267  while (nevents > 0 && ! errorflag)
268  {
269  evt = eventiterator->getNextEvent();
270  if (! evt)
271  {
272  cout << "Error after skipping " << i - nevents

274  errorflag = -1;
275  fileclose();
276  }
277  else
