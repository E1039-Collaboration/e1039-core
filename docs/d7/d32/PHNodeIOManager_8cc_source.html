190 PHNodeIOManager::write(TObject** data, const string& path)
191 {
192  if (file && tree)
193  {
194  TBranch *thisBranch = tree->GetBranch(path.c_str());
195  if (!thisBranch)
196  {
197  // Here is were we decide how to save the data in the root
198  // tree. split=0(prior to Root3.01/05; needs -1 afterwards

289 PHNodeIOManager::getBranchClassName(TBranch* branch)
290 {
291  // OK. Here all the game is to find out the name of the type
292  // contained in this branch. In ROOT pre-3.01/05 versions, all
293  // branches we used were of the same type = TBranchObject, so that
294  // was easy. Since version 3.01/05 ROOT introduced new branch style
295  // with some TBranchElement objects. So far so good. The problem is
296  // that I did not find a common way to grab the typename of the
297  // object contained in those branches, so I hereby use some durty if
298  // { } else if { } ...
299 
300 #if ROOT_VERSION_CODE >= ROOT_VERSION(3,01,5)
301  TBranchElement* be = dynamic_cast<TBranchElement*>(branch);
302 
303  if (be)
304  {
305  // TBranchElement has a nice GetClassName() method for us :
306  return be->GetClassName();
307  }
308 #endif
309 
310  TBranchObject* bo = dynamic_cast<TBranchObject*>(branch);
311  if (bo)
312  {
313  // For this one we need to go down a little before getting the
314  // name...
315  TLeafObject* leaf = static_cast<TLeafObject*>
316  (branch->GetLeaf(branch->GetName()));
317  assert(leaf != 0);
318  return leaf->GetTypeName();
319  }
320  cout << PHWHERE << "Fatal error, dynamic cast of TBranchObject failed" << endl;
321  exit(1);

398 PHCompositeNode*
399 PHNodeIOManager::reconstructNodeTree(PHCompositeNode* topNode)
400 {
401  if (! file)
402  {
403  if (filename.empty())
404  {
405  cout << PHWHERE << "filename was never set" << endl;
406  }
407  else
408  {
409  cout << PHWHERE << "TFile " << filename << " NULL pointer" << endl;
410  }
411  return NULL;
412  }
413 
414  tree = static_cast<TTree*>(file->Get(TreeName.c_str()));
415 
416  if (!tree)
417  {
418  cout << PHWHERE << "PHNodeIOManager::reconstructNodeTree : Root Tree "
419  << TreeName << " not found in file " << file->GetName() << endl;
420  return NULL;
421  }
422 
423  // ROOT sucks, we need a unique name for the tree so we can open multiple
424  // files. So we take the memory location of the file pointer which
425  // should be unique within this process to create it
426  ostringstream nname;
427  nname << TreeName << file;
428 
429 
430  tree->SetName(nname.str().c_str());
431 
432  // Select the branches according to objectToRead
433  map<string, PHBoolean>::const_iterator it;
434 
435  if (tree->GetNbranches() > 0)
436  {
437  for (it = objectToRead.begin(); it != objectToRead.end(); ++it)
438  {
439  tree->SetBranchStatus((it->first).c_str(),
440  static_cast<bool>(it->second));
441  }
442  }
443  // The file contains a TTree with a list of the TBranchObjects
444  // attached to it.
445  TObjArray *branchArray = tree->GetListOfBranches();
446 
447  // We need these in the loops down below...
448  size_t i, j;
449 
450  // If a topNode was provided, we can feed the iterator with it.
451  if (!topNode)
452  {
453  topNode = new PHCompositeNode("TOP"); // create topNode if we got a null pointer
454  }
455  PHNodeIterator nodeIter(topNode);
456 
457  // Loop over all branches in the tree. Each branch-name contains the
458  // full 'path' of composite-nodes in the original node tree. We
459  // split the name and reconstruct the tree.
460  string delimeters = phooldefs::branchpathdelim + "/"; // add old backslash for backward compat
461  for (i = 0; i < (size_t)(branchArray->GetEntriesFast()); i++)
462  {
463  string branchname = (*branchArray)[i]->GetName();
464  vector<string> splitvec;
465  boost::split(splitvec, branchname, boost::is_any_of(delimeters));
466  for (size_t ia = 1; ia< splitvec.size()-1; ia++) // -1 so we skip the node name
467  {
468  if (!nodeIter.cd(splitvec[ia]))
469  {
470  nodeIter.addNode(new PHCompositeNode(splitvec[ia]));
471  nodeIter.cd(splitvec[ia]);
472  }
473  }
474  TBranch *thisBranch = (TBranch*)((*branchArray)[i]);
475 
476  // Skip non-selected branches
477  if (thisBranch->TestBit(kDoNotProcess))
478  {
479  continue;
480  }
481 
482  string branchClassName = getBranchClassName(thisBranch);
483  string branchName = thisBranch->GetName();
484  fBranches[branchName] = thisBranch;
485 
486  assert(gROOT != 0);
487  TClass* thisClass = gROOT->GetClass(branchClassName.c_str());
488 
489  if (!thisClass)
490  {
491  cout << PHWHERE << endl;
492  cout << "Missing Class: " << branchClassName.c_str() << endl;
493  cout << "Did you forget to load the shared library which contains "
494  << branchClassName.c_str() << "?" << endl;
495  }
496  // it does not make sense to continue - the code coredumps
497  // later if a class is not loaded
498  assert(thisClass != 0);
499 
500  PHIODataNode<TObject> *newIODataNode =
501  static_cast<PHIODataNode<TObject> *> (nodeIter.findFirst("PHIODataNode", (*splitvec.rbegin()).c_str()));
502  if (! newIODataNode)
503  {
504  TObject *newTObject = static_cast<TObject*>(thisClass->New());
505  newIODataNode = new PHIODataNode<TObject>(newTObject, (*splitvec.rbegin()).c_str());
506  nodeIter.addNode(newIODataNode);
507  }
508  else
509  {
510  TObject *oldobject = newIODataNode->getData();
511  string oldclass = oldobject->ClassName();
512  if (oldclass != branchClassName)
513  {
514  cout << "You only have to worry if you get this message when reading parallel files"
515  << endl
516  << "if you get this when opening the 2nd, 3rd,... file" << endl
517  << "It looks like your objects are not of the same version in these files" << endl;
518  cout << PHWHERE << "Found object " << oldobject->ClassName()
519  << " in node tree but the file "
520  << filename << " contains a " << branchClassName
521  << " object. The object will be replaced without harming you" << endl;
522  cout << "CAVEAT: If you use local copies of pointers to data nodes" << endl
523  << "instead of searching the node tree you are in trouble now" << endl;
524  delete newIODataNode;
525  TObject *newTObject = static_cast<TObject*>(thisClass->New());
526  newIODataNode = new PHIODataNode<TObject>(newTObject, (*splitvec.rbegin()).c_str());
527  nodeIter.addNode(newIODataNode);
528  }
529 
530  }
531 
532  if (thisClass->InheritsFrom("PHObject"))
533  {
534  newIODataNode->setObjectType("PHObject");
535  }
536  else
537  {
538  cout << PHWHERE << branchClassName.c_str()
539  << " inherits neither from PHTable nor from PHObject"
540  << " setting type to PHObject" << endl;
541  newIODataNode->setObjectType("PHObject");
542  }
543  thisBranch->SetAddress(&(newIODataNode->data));
544  for (j = 1; j < splitvec.size() - 1; j++)
545  {
546  nodeIter.cd("..");
547  }
548 
549  }
550  return topNode;
551 }
